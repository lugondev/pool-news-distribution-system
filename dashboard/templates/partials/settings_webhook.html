Payload Format

// POST mode: Content-Type: application/json (default)
{
  "id": "a1b2c3d4e5f6...",
  "source_id": "techcrunch",
  "source_name": "TechCrunch",
  "url": "https://techcrunch.com/...",
  "title": "Original article title",
  "lang": "en",
  "category": "tech",
  "published_at": "2026-03-20T10:30:00+00:00",
  "summary": {
    "vi": "Tóm tắt tiếng Việt...",
    "en": "English summary..."
  },
  "sent_at": "2026-03-20T10:35:00+00:00"
}

// GET mode: sends as query parameters
// Template mode: renders custom Jinja2 template (POST = raw text body, GET = ?message=...)
