API Usage

# Fetch new articles from a channel
curl -H "X-API-Key: YOUR_API_KEY" \
  -H "X-Client-ID: my-bot-1" \
  "http://localhost:8000/api/channels/{id}/feed"

# Pick ONE article to post now (never repeats)
curl -H "X-API-Key: YOUR_API_KEY" \
  -H "X-Client-ID: my-bot-1" \
  "http://localhost:8000/api/channels/{id}/next"

# With query parameters
curl -H "X-API-Key: YOUR_API_KEY" \
  -H "X-Client-ID: my-bot-1" \
  "http://localhost:8000/api/channels/{id}/feed?limit=10"

# Acknowledge received articles (advance cursor)
curl -X POST -H "X-API-Key: YOUR_API_KEY" \
  -H "X-Client-ID: my-bot-1" \
  -H "Content-Type: application/json" \
  -d '{"cursor": "2025-01-15T12:00:00Z"}' \
  "http://localhost:8000/api/channels/{id}/ack"
